def pivot(list)
  pivoted = {}
  list.each do |item|
    value = yield(item)
  
    if value.is_a?(Array)
      if value.empty?
        pivoted[nil] ||= []
        pivoted[nil] << item 
      else
        value.each do |val|
          pivoted[val] ||= []
          pivoted[val] << item 
        end
      end
    else
      pivoted[value] ||= []
      pivoted[value] << item 
    end
  end

  pivoted
end
